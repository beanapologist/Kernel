\subsection{Theorem~11 — Coherence Function}

\begin{theorem}[Coherence function]\label{thm:11}
  The $\ell^1$ coherence as a function of $r>0$ is
  \begin{equation}
    C(r) \;=\; \frac{2r}{1+r^2}.
    \label{eq:coherence}
  \end{equation}
  Properties:
  \begin{enumerate}
    \item $C(1) = 1$ (global maximum).
    \item $C(r) = C(1/r)$ for all $r>0$ (symmetric about $r=1$).
    \item $\frac{dC}{dr}\big|_{r=1} = 0$,\quad
          $\frac{d^2C}{dr^2}\big|_{r=1} = -1 < 0$ (strict local maximum).
    \item $0 < C(r) < 1$ for all $r\ne 1$.
  \end{enumerate}
\end{theorem}

\begin{proof}
  Writing $C(r)=2r/(1+r^2)$:
  \begin{itemize}
    \item $C(1) = 2/2 = 1$.
    \item $C(1/r) = (2/r)/(1+1/r^2) = (2/r)\cdot r^2/(r^2+1) = 2r/(1+r^2) = C(r)$.
    \item $C'(r) = [2(1+r^2) - 2r\cdot 2r]/(1+r^2)^2 = 2(1-r^2)/(1+r^2)^2$;
          at $r=1$: $C'(1)=0$.
    \item $C''(r) = -4r(3-r^2)/(1+r^2)^3$; at $r=1$: $C''(1)=-4/4=-1<0$. \qed
  \end{itemize}
\end{proof}

\end{lemma}

\begin{proof}

  $\sin(\Delta\Phi/2) \approx \pi/\sqrt{n}$ and

\begin{equation}

                   = \sum_{i=1}^{N} \operatorname{sech}(\lambda_i),\qquad
  R_{\mathrm{tot}} = \frac{1}{G_{\mathrm{tot}}}.

\end{equation}
For $N$ identical channels ($\lambda_i = \lambda$):
\begin{equation}

  R_{\mathrm{tot}} = \frac{\cosh\lambda}{N}.

\begin{equation}
  R_{\mathrm{tot}} = \sum_{j=1}^{M} \Reff(\lambda_j)

  G_{\mathrm{tot}} = \frac{1}{R_{\mathrm{tot}}}.

The bottleneck stage is $j^* = \operatorname{argmax}_j \cosh(\lambda_j)$.

    g_{\mathrm{fb}} = g\cdot\alpha_{\mathrm{adaptive}}\cdot R_{\mathrm{mid}},

  \text{Sub-step 2:} &\quad
    \hat\psi_j \leftarrow \hat\psi_j - g_{\mathrm{fb}}\,\delta\theta_j,
    \label{eq:fb-sub2}
\end{align}
where $\Delta E = E_{\mathrm{before}} - E_{\mathrm{mid}}$ (frustration decay
in sub-step 1), $\Delta R = R_{\mathrm{mid}} - R_{\mathrm{before}}$ (coherence
gain in sub-step 1), $R_{\mathrm{mid}}$ is the circular coherence after
sub-step 1, and $c_1, c_2 \ge 0$ are sensitivity parameters
(default $c_1 = c_2 = 0$, reproducing fixed-$\alpha$ behaviour).

  For $g \in (0,1]$ and $\alpha_{\mathrm{adaptive}} \in (0,1]$:
  \begin{enumerate}
    \item $g_{\mathrm{fb}} = g\cdot\alpha_{\mathrm{adaptive}}\cdot R_{\mathrm{mid}}
          \le g \le 1$, so both sub-steps are dissipative contractions.
    \item \texttt{feedback\_step} converges at least as fast as \texttt{step}.
    \item $R = 1$ is a fixed point: zero frustration implies all
          $\delta\theta_j = 0$, leaving both sub-steps inert.
  \end{enumerate}
\end{proposition}

\begin{proof}
  Since $R_{\mathrm{mid}}\in[0,1]$, $\alpha_{\mathrm{adaptive}}\le1$, and
  $g\le1$, the product $g_{\mathrm{fb}}\le1$.  Each sub-step is therefore a
  contraction with factor $\abs{1-g_{\mathrm{fb}}}<1$ (resp.\ $\abs{1-g}<1$).
  The second sub-step removes residual frustration left after sub-step 1;
  hence total convergence is at least as fast.  At $R=1$ all phases equal
  $\bar\psi$, so $\delta\theta_j=0$ for all $j$ and neither sub-step moves any
  phase. \qed
\end{proof}

\end{proposition}

\begin{theorem}[$\Theta(\sqrt{n})$ search time]\label{thm:main-sqrt}

\begin{proof}

  \hat\psi \;\leftarrow\;
  \hat\psi + \alpha_\psi\,\operatorname{wrap}(\hat\psi_{\mathrm{new}} - \hat\psi).
  \label{eq:ks-ema-carrier}
\end{equation}

\subsection{Grover Analogy: EMA as Damped Diffusion}
% Full derivation in experiments/kernelsync_demo/grover_analogy.md.

\begin{proposition}[KernelSync EMA as damped Grover diffusion]%
  \label{prop:ks-grover}
  Let $\delta\theta_j = \hat\psi_j - \bar\psi$ be the phase deviation of
  follower node $j$ from the collective mean $\bar\psi$.  The carrier-phase
  EMA update with gain $g = \alpha_\psi$ satisfies, to first order in
  $\delta\theta_j$:
  \begin{equation}
    \delta\theta_j^{(t+1)} \;\approx\; (1-g)\,\delta\theta_j^{(t)}.
    \label{eq:ks-ema}
  \end{equation}
  For $g = 1$ this reduces exactly to the Grover diffusion step
  $a_x \mapsto 2\bar{a} - a_x$.
\end{proposition}

\begin{proof}
  The coherent receiver residual is
  $\varphi_{\mathrm{res},j} \approx \bar\psi - \hat\psi_j = -\delta\theta_j$
  (small-angle approximation, valid for $|\delta\theta_j| \ll \pi$).
  One EMA step gives
  $\hat\psi_j^{\mathrm{new}} = \hat\psi_j + g(-\delta\theta_j)$.
  The collective mean is invariant to first order
  ($\sum_j \varphi_{\mathrm{res},j} \approx -\sum_j \delta\theta_j = 0$),
  so
  $\delta\theta_j^{\mathrm{new}}
   = \hat\psi_j^{\mathrm{new}} - \bar\psi = (1-g)\,\delta\theta_j$. \qed
\end{proof}

\noindent The matched-filter phase-minimum selection plays the role of the
oracle $O = I - 2\ket{x^*}\bra{x^*}$: only the true timing offset produces a
correlated (low-residual-phase) peak, marking it before the diffusion step
contracts all deviations by $(1-g)^k$ over $k$ iterations.

\begin{remark}[Conserved quantities]
  The exact Grover diffusion ($g=1$) conserves the total squared norm
  $\sum_j\delta_j^2$ (it is an isometry).  The KernelSync EMA ($g<1$) does
  not: $\sum_j(\delta_j^{\mathrm{new}})^2 = (1-g)^2\sum_j\delta_j^2$, so
  phase consensus is achieved dissipatively.  The one quantity conserved by
  both is the mean phase $\bar\psi$.
\end{remark}

\subsection{Experimental Goal}

\subsection{Phase Schedule}

For step $n$ (0-indexed), with chip period $T_{\mathrm{step}} = 40$\,ns at
sample rate $f_s = 2.4$\,GS/s (96 identical samples per step):
\begin{align}
  \varphi_\mu(n)    &= \tfrac{3\pi}{4}\,(n\bmod 8),
    \label{eq:ghz-fast}\\
  \varphi_p(n)      &= \tfrac{2\pi n}{D},
    \label{eq:ghz-slow}\\
  \varphi(n)        &= \varphi_\mu(n) + \varphi_p(n)
    \quad (\text{palindromic variant}),
    \label{eq:ghz-pal}\\
  \varphi_{\mathrm{ctrl}}(n) &= \varphi_\mu(n)
    \quad (\text{control variant — no precession}).
    \label{eq:ghz-ctrl}
\end{align}
The baseband IQ envelope for step $n$ is
\begin{equation}
  s[n] = A\,e^{i\varphi(n)},
  \qquad
  I[n] = A\cos\varphi(n),\quad
  Q[n] = A\sin\varphi(n).
  \label{eq:ghz-iq}
\end{equation}
The constant envelope $|s[n]| = A$ mirrors the unit-norm phasor
$|P(n)| = 1$ of the palindrome precession (Theorem~\ref{thm:zero-overhead}).

\subsection{Measurement Protocol}

Three sequential runs isolate the effect of the precession:
\begin{enumerate}
  \item \textbf{Baseline ringdown:} drive off; record free ringdown and fit
        $\kappa_{\mathrm{baseline}}$.
  \item \textbf{Palindromic drive:} apply $\varphi(n)$ for $N_{\mathrm{steps}}$;
        fit $\kappa_{\mathrm{after\,pal}}$ from the post-drive ringdown.
  \item \textbf{Control drive:} apply $\varphi_{\mathrm{ctrl}}(n)$ for the
        same duration; fit $\kappa_{\mathrm{after\,ctrl}}$ from the
        post-drive ringdown.
\end{enumerate}
The figure of merit is the \emph{differential decay shift}:
\begin{equation}
  \Delta\kappa
  = \kappa_{\mathrm{after\,pal}} - \kappa_{\mathrm{after\,ctrl}},
  \label{eq:delta-kappa}
\end{equation}
estimated with bootstrap error.  A non-zero $\Delta\kappa$ would indicate
that the slow palindromic precession (period $D$ steps, angular increment
$\Delta\Phi_0 = 2\pi/D$ rad/step) modifies the effective dissipation of the
resonator, providing a direct physical signature of the two-frequency torus
geometry $T^2 = S^1\times S^1$ in a superconducting circuit.
